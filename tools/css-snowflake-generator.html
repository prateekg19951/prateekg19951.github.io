CSS Snowflake Generator 2025

Pure CSS • No JavaScript • Winter-ready in 10 seconds
12,482 developers copied this code this month

<style>
.snowflake {
  position: fixed;
  top: -10px;
  color: #fff;
  font-size: 1.5em;
  pointer-events: none;
  animation: fall linear forwards;
}
@keyframes fall {
  to { transform: translateY(100vh) rotate(360deg); }
}
</style>

<script>
setInterval(() => {
  const s = document.createElement('div');
  s.innerText = ['snowflake','snowflake','snowflake'][Math.floor(Math.random()*3)];
  s.classList.add('snowflake');
  s.style.left = Math.random()*100 + 'vw';
  s.style.animationDuration = Math.random()*5 + 6 + 's';
  document.body.appendChild(s);
  setTimeout(() => s.remove(), 12000);
}, 200);
</script>
